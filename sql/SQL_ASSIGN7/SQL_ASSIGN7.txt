CREATE TABLE COLLEGE(
COLLEGEID NUMBER (8) NOT NULL,
CNAME CHAR (15) NOT NULL,
CONSTRAINT COLLEGE_PRIMARY_KEY PRIMARY KEY (COLLEGEID)
);

CREATE TABLE LOCATION(
LID NUMBER (8) NOT NULL,
LOC CHAR (4),
CONSTRAINT LOCATION_PRIMARY_KEY PRIMARY KEY (LID)
);

CREATE TABLE STATUS(
SID NUMBER (8) NOT NULL,
STATNAME CHAR (24),
CONSTRAINT STATUS_PRIMARY_KEY PRIMARY KEY (SID)
);

CREATE TABLE USER(
ID NUMBER (8) NOT NULL,
NAME CHAR (24) NOT NULL,
EMAIL CHAR (21) NOT NULL,
ADDRESS CHAR (45) NOT NULL,
PHONE NUMBER (11) NOT NULL,
SID NUMBER (8) NOT NULL,
COLLEGEID NUMBER (8) NOT NULL,
CONSTRAINT STUDENT_PRIMARY_KEY PRIMARY KEY (ID),
CONSTRAINT STUDENT_FOREIGN_KEY FOREIGN KEY (SID) REFERENCES STATUS (SID),
CONSTRAINT STUDENT2_FOREIGN_KEY FOREIGN KEY (COLLEGEID) REFERENCES COLLEGE (COLLEGEID)
);

CREATE TABLE PAYMENT(
PID NUMBER (8) NOT NULL,
PRICE NUMBER (7) NOT NULL,
ID NUMBER (8) NOT NULL,
CONSTRAINT PAYMENT_PRIMARY_KEY PRIMARY KEY (PID),
CONSTRAINT PAYMENT_FOREIGN_KEY FOREIGN KEY (ID) REFERENCES STUDENT (ID)
);

CREATE TABLE DEPARTMENT(
DID NUMBER (4) NOT NULL,
DNAME CHAR (10),
LID NUMBER (8) NOT NULL,
COLLEGEID NUMBER (8) NOT NULL,
CONSTRAINT DEPARTMENT_PRIMARY_KEY PRIMARY KEY (DID),
CONSTRAINT DEPARTMENT_FOREIGN_KEY FOREIGN KEY (LID) REFERENCES LOCATION (LID),
CONSTRAINT DEPARTMENT1_FOREIGN_KEY FOREIGN KEY (COLLEGEID) REFERENCES COLLEGE (COLLEGEID)
);

CREATE TABLE COURSE(
COID NUMBER (4) NOT NULL,
CNAME CHAR (20) NOT NULL,
DID NUMBER (4) NOT NULL,
CONSTRAINT COURSE_PRIMARY_KEY PRIMARY KEY (COID),
CONSTRAINT COURSE_FOREIGN_KEY FOREIGN KEY (DID) REFERENCES DEPARTMENT (DID)
);

CREATE TABLE SECTION(
SECID NUMBER (4) NOT NULL,
SECNUM NUMBER (2),
LID NUMBER (8) NOT NULL,
COID NUMBER (4),
STIME CHAR (5),
ETIME CHAR (5),
ID NUMBER (8) NOT NULL,
CONSTRAINT SECTION_PRIMARY_KEY PRIMARY KEY (SECID),
CONSTRAINT SECTION_FOREIGN_KEY FOREIGN KEY (LID) REFERENCES LOCATION (LID),
CONSTRAINT SECTION2_FOREIGN_KEY FOREIGN KEY (COID) REFERENCES COURSE (COID),
CONSTRAINT SECTION3_FOREIGN_KEY FOREIGN KEY (ID) REFERENCES STUDENT (ID)
);

CREATE TABLE CATALOG(
CID NUMBER (4) NOT NULL,
DID NUMBER (4) NOT NULL,
SECID NUMBER (4) NOT NULL,
LID NUMBER (8) NOT NULL,
ID NUMBER (8) NOT NULL,
CONSTRAINT CATALOG_PRIMARY_KEY PRIMARY KEY (CID),
CONSTRAINT STATUS_FOREIGN_KEY FOREIGN KEY (DID) REFERENCES DEPARTMENT (DID),
CONSTRAINT STATUS1_FOREIGN_KEY FOREIGN KEY (SECID) REFERENCES SECTION (SECID),
CONSTRAINT STATUS2_FOREIGN_KEY FOREIGN KEY (LID) REFERENCES LOCATION (LID),
CONSTRAINT STATUS3_FOREIGN_KEY FOREIGN KEY (ID) REFERENCES STUDENT (ID)
);

CREATE TABLE STUREG(
STUREGID NUMBER (8) NOT NULL,
PID NUMBER (8) NOT NULL,
SECID NUMBER (8) NOT NULL,
ID NUMBER (8) NOT NULL,
CONSTRAINT STUREG_PRIMARY_KEY PRIMARY KEY (STUREGID),
CONSTRAINT STUREG_FOREIGN_KEY FOREIGN KEY (PID) REFERENCES PAYMENT (PID),
CONSTRAINT STUREG1_FOREIGN_KEY FOREIGN KEY (SECID) REFERENCES SECTION (SECID),
CONSTRAINT STUREG3_FOREIGN_KEY FOREIGN KEY (ID) REFERENCES STUDENT (ID)
);
