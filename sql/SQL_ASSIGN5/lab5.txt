select ENAME,
    SAL,
    lpad('*',trunc(SAL/100),'*') EMPLOYEE_AND_THEIR_SALARIES
    FROM EMP;

 SELECT ENAME AS NAME, SUBSTR(LOWER(ENAME), 1, 2) AS USERNAME, HIREDATE AS HIREDATE, ROUND(SAL) AS SALARY, TO_CHAR(ADD_MONTHS(HIREDATE, 6), 'MM/DD/YY') AS REVIEW
    FROM EMP;



SELECT ENAME, EMPNO, SAL FROM EMP
    WHERE SAL > 1000 AND DEPTNO IN
    (SELECT DEPTNO FROM DEPT
    WHERE LOC LIKE 'BOSTON%' AND DEPTNO LIKE '40%');



 SELECT HIREDATE
    FROM EMP
    WHERE HIREDATE >='01-JAN-81' AND HIREDATE <='31-DEC-81';


//CORRELAED SUBQUERY
SELECT * FROM EMP
    WHERE DEPTNO = ANY (SELECT DEPTNO FROM DEPT
    WHERE DEPT.DEPTNO = EMP.DEPTNO);


//INLINE VIEW
 SELECT *
    FROM ( SELECT ENAME, EMPNO, DEPTNO
    FROM EMP
    ) EMP, DEPT
    WHERE DEPT.DEPTNO = EMP.DEPTNO;
