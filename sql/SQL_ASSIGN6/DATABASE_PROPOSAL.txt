CREATE TABLE STUDENT(
STUID NUMBER(8) NOT NULL,
NAME CHAR(24) NOT NULL,
EMAIL CHAR (21),
STATUS CHAR (2),
DEPTNO NUMBER(4) NOT NULL,
CONSTRAINT STUDENT_FOREIGN_KEY FOREIGN KEY (DEPTNO) REFERENCES
DEPARTMENT (DEPTNO),
CONSTRAINT STUDENT_PRIM_KEY PRIMARY KEY (STUID)
);

CREATE TABLE DEPARTMENT(
DEPTNO NUMBER(4) NOT NULL,
LOC CHAR (6) NOT NULL,
CONSTRAINT DEPARTMENT_PRIMARY_KEY PRIMARY KEY (DEPTNO)
);


CREATE TABLE COURSE(
STUID NUMBER(8) NOT NULL,
CRN NUMBER(4) NOT NULL,
NAME CHAR (15) NOT NULL,
DEPTNO NUMBER(4) NOT NULL,
PRICE NUMBER (5) NOT NULL,
UNITS NUMBER (2),
CONSTRAINT COURSE_FOREIGN_KEY FOREIGN KEY (STUID) REFERENCES
STUDENT (STUID),
CONSTRAINT COURSE1_FOREIGN_KEY FOREIGN KEY (DEPTNO) REFERENCES
DEPARTMENT (DEPTNO),
CONSTRAINT COURSE_PRIM_KEY PRIMARY KEY (NAME)
);

CREATE TABLE SCHEDULE(
STUID NUMBER(8) NOT NULL,
NAME CHAR(15) NOT NULL,
PROFNAME CHAR(24) NOT NULL,
STIME CHAR(5),
ETIME CHAR(5),
DEPTNO NUMBER(4) NOT NULL,
DAYS CHAR (7) NOT NULL,
CONSTRAINT SCHEDULE_FOREIGN_KEY FOREIGN KEY (NAME) REFERENCES
COURSE (NAME),
CONSTRAINT SCHEDULE1_FOREIGN_KEY FOREIGN KEY (DEPTNO) REFERENCES
DEPARTMENT (DEPTNO)
);
